    // ===== РАСЧЁТ СТОИМОСТИ =====
    function calculateServiceCost(service, settings) {
      const st = serviceTypes.find(s =&gt; s.id === service.type) || serviceTypes[0];
      const level = levels.find(l =&gt; l.id === service.level) || levels[2];
      const qual = qualificationLevels.find(q =&gt; q.id === service.qualification) || qualificationLevels[0];
      const pg = pflegegradLevels.find(p =&gt; p.id === service.pflegegrad) || pflegegradLevels[0];
      const region = regions.find(r =&gt; r.id === settings.regionId) || regions[0];

      const base = st.baseRate;
      const factor = level.factor * qual.factor * pg.factor * region.factor;

      // Базовая ставка с коэффициентами
      let baseHourly = base * factor;

      // Надбавки за время (приблизительный метод: доплата как доля базовой)
      const tf = service.timeFactors || {};
      let tfMultiplier = 1;
      if (tf.evening) tfMultiplier += 0.10;
      if (tf.night) tfMultiplier += 0.25;
      if (tf.weekend) tfMultiplier += 0.15;
      if (tf.holiday) tfMultiplier += 0.35;

      baseHourly *= tfMultiplier;

      // Спец. потребности (например, необходимость 2-х работников, особые риски, сложные задачи)
      if (service.specialNeeds) {
        baseHourly *= 1.15;
      }

      // Дополнительные расходы (процентом от зарплатной части)
      const social = settings.socialContrib;
      const vac = settings.vacationSick;
      const admin = settings.admin;
      const ap = settings.additionalPayments;

      // Транспорт/прочие фиксированные из региона и настроек
      const fixed = (settings.fixedPerHour || 0) + (region.transportPerHour || 0);

      // Полная ставка в час (модель: базовая * (1 + проценты) + фикс)
      const hourlyCost = baseHourly * (1 + social + vac + admin + ap) + fixed;

      // Стоимость услуги
      const total = hourlyCost * Math.max(0, parseFloat(service.hours) || 0);

      return {
        base,
        factor,
        tfMultiplier,
        hourlyCost,
        total,
        breakdown: {
          baseHourlyBeforeOverheads: baseHourly,
          social: baseHourly * social,
          vacationSick: baseHourly * vac,
          admin: baseHourly * admin,
          additional: baseHourly * ap,
          fixed,
        }
      };
    }

    function generateAIRecommendations(services, settings) {
      const recs = [];
      const totals = services.map(s =&gt; calculateServiceCost(s, settings).total);
      const grandTotal = totals.reduce((a, b) =&gt; a + b, 0);

      // 1) Предупреждение о высоких суммах
      if (grandTotal &gt; settings.highCostWarn) {
        recs.push({
          severity: &#x27;warn&#x27;,
          text: `Сумма по всем услугам ${fmtMoney(grandTotal)} превышает порог ${fmtMoney(settings.highCostWarn)}. Проверьте часы или сложность.`
        });
      }

      // 2) Слияние мелких услуг одного типа
      const byType = services.reduce((acc, s) =&gt; {
        acc[s.type] = acc[s.type] || { hours: 0, rows: [] };
        acc[s.type].hours += Number(s.hours) || 0;
        acc[s.type].rows.push(s);
        return acc;
      }, {});
      Object.entries(byType).forEach(([type, info]) =&gt; {
        if (info.rows.length &gt;= 2 &amp;&amp; info.rows.every(r =&gt; r.level === info.rows[0].level &amp;&amp; r.qualification === info.rows[0].qualification)) {
          recs.push({
            severity: &#x27;info&#x27;,
            text: `Есть несколько позиций &quot;${serviceTypes.find(t =&gt; t.id === type)?.label || type}&quot; с одинаковыми параметрами. Объедините их для упрощения отчётности (итого часов: ${fmtNumber(info.hours)}).`
          });
        }
      });

      // 3) Ночные/праздничные надбавки
      services.forEach(s =&gt; {
        const tf = s.timeFactors || {};
        const flags = [&#x27;night&#x27;, &#x27;holiday&#x27;].filter(k =&gt; tf[k]);
        if (flags.length) {
          recs.push({
            severity: &#x27;tip&#x27;,
            text: `Для услуги &quot;${serviceTypes.find(t =&gt; t.id === s.type)?.label}&quot; учтены повышающие коэффициенты: ${flags.join(&#x27;, &#x27;)}. При возможности перенесите часть часов на обычное время.`
          });
        }
      });

      // 4) Региональные затраты
      const region = regions.find(r =&gt; r.id === settings.regionId);
      if (region &amp;&amp; region.transportPerHour &gt; 0) {
        recs.push({
          severity: &#x27;note&#x27;,
          text: `В регионе &quot;${region.label}&quot; добавлен транспорт/прочие фиксированные расходы: ~${fmtMoney(region.transportPerHour)} за час.`
        });
      }

      return { grandTotal, recs };
    }

    function TextExport({ services, settings }) {
      const buildText = () =&gt; {
        const lines = [];
        lines.push(&#x27;СМЕТА УСЛУГ&#x27;);
        lines.push(&#x27;================&#x27;);
        const region = regions.find(r =&gt; r.id === settings.regionId);
        lines.push(`Регион: ${region?.label || settings.regionId}`);
        lines.push(`Параметры: соц.взносы ${(settings.socialContrib*100).toFixed(1)}%, отпуск/больничные ${(settings.vacationSick*100).toFixed(1)}%, админ ${(settings.admin*100).toFixed(1)}%, доплаты ${(settings.additionalPayments*100).toFixed(1)}%, фикс €/ч ${settings.fixedPerHour}`);
        lines.push(&#x27;&#x27;);

        let grand = 0;
        services.forEach((s, i) =&gt; {
          const st = serviceTypes.find(t =&gt; t.id === s.type);
          const c = calculateServiceCost(s, settings);
          grand += c.total;
          lines.push(`${i+1}. ${st?.label || s.type}`);
          lines.push(`   Часы: ${s.hours}`);
          lines.push(`   Ставка, €/ч: ${c.hourlyCost.toFixed(2)}`);
          lines.push(`   Сумма: ${c.total.toFixed(2)} €`);
          if (s.note) lines.push(`   Примечание: ${s.note}`);
          lines.push(&#x27;&#x27;);
        });

        lines.push(`ИТОГО: ${grand.toFixed(2)} €`);
        return lines.join(&#x27;\n&#x27;);
      };

      const download = () =&gt; {
        const text = buildText();
        const blob = new Blob([text], { type: &#x27;text/plain;charset=utf-8&#x27; });
        const url = URL.createObjectURL(blob);
        const a = document.createElement(&#x27;a&#x27;);
        a.href = url;
        a.download = &#x27;smeta.txt&#x27;;
        a.click();
        URL.revokeObjectURL(url);
      };

    function calculateServiceCost(service, settings) {
      const st = serviceTypes.find(s =&gt; s.id === service.type) || serviceTypes[0];
      const level = levels.find(l =&gt; l.id === service.level) || levels[2];
      const qual = qualificationLevels.find(q =&gt; q.id === service.qualification) || qualificationLevels[0];
      const pg = pflegegradLevels.find(p =&gt; p.id === service.pflegegrad) || pflegegradLevels[0];
      const region = regions.find(r =&gt; r.id === settings.regionId) || regions[0];

      const base = st.baseRate;
      const factor = level.factor * qual.factor * pg.factor * region.factor;

      // Базовая ставка с коэффициентами
      let baseHourly = base * factor;

      // Надбавки за время
      const tf = service.timeFactors || {};
      let tfMultiplier = 1;
      if (tf.evening) tfMultiplier += 0.10;
      if (tf.night) tfMultiplier += 0.25;
      if (tf.weekend) tfMultiplier += 0.15;
      if (tf.holiday) tfMultiplier += 0.35;
      baseHourly *= tfMultiplier;

      // Особые потребности
      if (service.specialNeeds) {
        baseHourly *= 1.15;
      }

      // Накладные
      const social = settings.socialContrib;
      const vac = settings.vacationSick;
      const admin = settings.admin;
      const ap = settings.additionalPayments;

      // Фиксированная часть (регион + общая)
      const fixed = (settings.fixedPerHour || 0) + (region.transportPerHour || 0);

      // Полная ставка в час
      const hourlyCost = baseHourly * (1 + social + vac + admin + ap) + fixed;

      // Стоимость услуги
      const total = hourlyCost * Math.max(0, parseFloat(service.hours) || 0);

      return {
        base,
        factor,
        tfMultiplier,
        hourlyCost,
        total,
        breakdown: {
          baseHourlyBeforeOverheads: baseHourly,
          social: baseHourly * social,
          vacationSick: baseHourly * vac,
          admin: baseHourly * admin,
          additional: baseHourly * ap,
          fixed,
        }
      };
    }

    // ===== РЕКОМЕНДАЦИИ ИИ (простые эвристики) =====
    function generateAIRecommendations(services, settings) {
      const recs = [];
      const totals = services.map(s =&gt; calculateServiceCost(s, settings).total);
      const grandTotal = totals.reduce((a, b) =&gt; a + b, 0);

      if (grandTotal &gt; settings.highCostWarn) {
        recs.push({
          severity: &#x27;warn&#x27;,
          text: `Сумма по всем услугам ${fmtMoney(grandTotal)} превышает порог ${fmtMoney(settings.highCostWarn)}. Проверьте часы или сложность.`
        });
      }

      // Слияние одинаковых строк одного типа/уровня/квалификации
      const byType = services.reduce((acc, s) =&gt; {
        acc[s.type] = acc[s.type] || { hours: 0, rows: [] };
        acc[s.type].hours += Number(s.hours) || 0;
        acc[s.type].rows.push(s);
        return acc;
      }, {});
      Object.entries(byType).forEach(([type, info]) =&gt; {
        if (info.rows.length &gt;= 2 &amp;&amp; info.rows.every(r =&gt; r.level === info.rows[0].level &amp;&amp; r.qualification === info.rows[0].qualification)) {
          recs.push({
            severity: &#x27;info&#x27;,
            text: `Есть несколько позиций &quot;${serviceTypes.find(t =&gt; t.id === type)?.label || type}&quot; с одинаковыми параметрами. Объедините их для упрощения отчётности (итого часов: ${fmtNumber(info.hours)}).`
          });
        }
      });

      // Ночные/праздничные надбавки
      services.forEach(s =&gt; {
        const tf = s.timeFactors || {};
        const flags = [&#x27;night&#x27;, &#x27;holiday&#x27;].filter(k =&gt; tf[k]);
        if (flags.length) {
          recs.push({
            severity: &#x27;tip&#x27;,
            text: `Для услуги &quot;${serviceTypes.find(t =&gt; t.id === s.type)?.label}&quot; учтены повышающие коэффициенты: ${flags.join(&#x27;, &#x27;)}. При возможности перенесите часть часов на обычное время.`
          });
        }
      });

      // Региональные фиксированные
      const region = regions.find(r =&gt; r.id === settings.regionId);
      if (region &amp;&amp; region.transportPerHour &gt; 0) {
        recs.push({
          severity: &#x27;note&#x27;,
          text: `В регионе &quot;${region.label}&quot; добавлен транспорт/прочие фиксированные расходы: ~${fmtMoney(region.transportPerHour)} за час.`
        });
      }

      return { grandTotal, recs };
    }

    function TextExport({ services, settings }) {
      const buildText = () =&gt; {
        const lines = [];
        lines.push(&#x27;СМЕТА УСЛУГ&#x27;);
        lines.push(&#x27;================&#x27;);
        const region = regions.find(r =&gt; r.id === settings.regionId);
        lines.push(\`Регион: \${region?.label || settings.regionId}\`);
        lines.push(\`Параметры: соц.взносы \${(settings.socialContrib*100).toFixed(1)}%, отпуск/больничные \${(settings.vacationSick*100).toFixed(1)}%, админ \${(settings.admin*100).toFixed(1)}%, доплаты \${(settings.additionalPayments*100).toFixed(1)}%, фикс €/ч \${settings.fixedPerHour}\`);
        lines.push(&#x27;&#x27;);

        let grand = 0;
        services.forEach((s, i) =&gt; {
          const st = serviceTypes.find(t =&gt; t.id === s.type);
          const c = calculateServiceCost(s, settings);
          grand += c.total;
          lines.push(\`\${i+1}. \${st?.label || s.type}\`);
          lines.push(\`   Часы: \${s.hours}\`);
          lines.push(\`   Ставка, €/ч: \${c.hourlyCost.toFixed(2)}\`);
          lines.push(\`   Сумма: \${c.total.toFixed(2)} €\`);
          if (s.note) lines.push(\`   Примечание: \${s.note}\`);
          lines.push(&#x27;&#x27;);
        });

        lines.push(\`ИТОГО: \${grand.toFixed(2)} €\`);
        return lines.join(&#x27;\\n&#x27;);
      };

      const download = () =&gt; {
        const text = buildText();
        const blob = new Blob([text], { type: &#x27;text/plain;charset=utf-8&#x27; });
        const url = URL.createObjectURL(blob);
        const a = document.createElement(&#x27;a&#x27;);
        a.href = url;
        a.download = &#x27;smeta.txt&#x27;;
        a.click();
        URL.revokeObjectURL(url);
      };

      return (
        &lt;div className=&quot;flex items-center gap-3&quot;&gt;
          &lt;button onClick={download} className=&quot;bg-emerald-600 hover:bg-emerald-700 text-white rounded-lg px-4 py-2&quot;&gt;
            Экспорт сметы (TXT)
          &lt;/button&gt;
          &lt;span className=&quot;text-sm text-gray-600&quot;&gt;Для печати используйте Ctrl/Cmd + P&lt;/span&gt;
        &lt;/div&gt;
      );
    }

    function App() {
      const [settings, setSettings] = useState(defaultSettings);
      const [services, setServices] = useState([ emptyService() ]);
      const [recsOpen, setRecsOpen] = useState(false);

      // Загрузка/сохранение состояния
      useEffect(() =&gt; {
        const raw = localStorage.getItem(STORAGE_KEY);
        if (!raw) return;
        try {
          const data = JSON.parse(raw);
          if (data.settings) setSettings({ ...defaultSettings, ...data.settings });
          if (Array.isArray(data.services) &amp;&amp; data.services.length) {
            // миграция полей
            setServices(data.services.map(s =&gt; ({
              ...emptyService(),
              ...s,
              timeFactors: { evening: false, night: false, weekend: false, holiday: false, ...(s.timeFactors || {}) }
            })));
          }
        } catch(e) {}
      }, []);

      useEffect(() =&gt; {
        const payload = JSON.stringify({ settings, services });
        localStorage.setItem(STORAGE_KEY, payload);
      }, [settings, services]);

      const addService = () =&gt; setServices(prev =&gt; [ ...prev, emptyService() ]);
      const removeService = (id) =&gt; setServices(prev =&gt; prev.filter(s =&gt; s.id !== id));
      const updateService = (id, key, value) =&gt; setServices(prev =&gt; prev.map(s =&gt; s.id === id ? ({ ...s, [key]: value }) : s));

      const calcRows = useMemo(() =&gt; services.map(s =&gt; ({
        id: s.id,
        s,
        calc: calculateServiceCost(s, settings)
      })), [services, settings]);

      const grandTotal = calcRows.reduce((a, r) =&gt; a + r.calc.total, 0);
      const doRecs = useMemo(() =&gt; generateAIRecommendations(services, settings), [services, settings]);

      return (

          &lt;header className=&quot;no-print flex items-start justify-between gap-3&quot;&gt;

            &lt;/div&gt;
            &lt;div className=&quot;flex gap-2&quot;&gt;
              &lt;button onClick={()=&gt;{ setServices([emptyService()]); }}
                      className=&quot;bg-white border border-gray-300 hover:bg-gray-50 rounded-lg px-3 py-2&quot;&gt;
                Очистить

              &lt;button onClick={()=&gt;window.print()}
                      className=&quot;bg-blue-600 hover:bg-blue-700 text-white rounded-lg px-3 py-2&quot;&gt;

                        className=&quot;bg-white border border-gray-300 hover:bg-gray-50 rounded-lg px-4 py-2&quot;&gt;

Да, **можно встроить как HTML-блок**:

## 🔧 Рекомендованные правки (очень коротко)

### 1) Явно указать пресеты Babel  
Чтобы JSX и современный JS стабильно компилировались в браузере, укажите пресеты:

**Было**:
```html
&lt;script type=&quot;text/babel&quot;&gt;
  // ваш React-код
&lt;/script&gt;
```

**Стало**:
```html
&lt;script type=&quot;text/babel&quot; data-presets=&quot;env,react&quot;&gt;
  // ваш React-код
&lt;/script&gt;
```

### 2) Перейти на `createRoot` для React 18  
Это уберёт предупреждения и согласует код с текущей рекомендацией React.

**Было** (самый низ файла):
```js
// Рендер приложения
ReactDOM.render(&lt;PersonalBudgetCalculator /&gt;, document.getElementById(&#x27;root&#x27;));
```

**Стало** (с безопасным фолбэком на старый рендер):
```js
// Рендер приложения (React 18 + фолбэк)
const container = document.getElementById(&#x27;root&#x27;);
const root = (ReactDOM.createRoot)
  ? ReactDOM.createRoot(container)
  : { render: (el) =&gt; ReactDOM.render(el, container) };

root.render(&lt;PersonalBudgetCalculator /&gt;);
```

                const regionFactor = regions[service.region].factor;
                
                const adjustedRate = baseRate * levelFactor * regionFactor;
                const baseCost = service.hours * adjustedRate;
                
                // Немецкие социальные взносы и надбавки
                const socialContributions = baseCost * 0.215; // 21.5%
                const vacationSick = (baseCost + socialContributions) * 0.342; // 34.2%
                const administration = (baseCost + socialContributions + vacationSick) * 0.10; // 10%
                const fixedCosts = service.hours * 3.5; // 3.5€ за час
                
                const totalCost = baseCost + socialContributions + vacationSick + administration + fixedCosts;
                
                return {
                    baseCost,
                    socialContributions,
                    vacationSick,
                    administration,
                    fixedCosts,
                    totalCost,
                    hourlyRate: totalCost / service.hours
                };
            };

            // Обновление общей стоимости
            useEffect(() =&gt; {
                const total = services.reduce((sum, service) =&gt; {
                    return sum + calculateServiceCost(service).totalCost;
                }, 0);
                setTotalCost(total);
                
                // Простые AI рекомендации
                generateSimpleRecommendations(total);
            }, [services]);

            // Простая логика AI рекомендаций
            const generateSimpleRecommendations = (total) =&gt; {
                const recommendations = [];
                
                if (total &gt; 3000) {
                    recommendations.push({
                        type: &#x27;warning&#x27;,
                        title: &#x27;Высокая стоимость&#x27;,
                        message: `Общая стоимость ${total.toFixed(2)}€ превышает региональные лимиты`,
                        action: &#x27;Рассмотрите сокращение часов&#x27;
                    });
                }
                
                if (services.length &gt; 3) {
                    recommendations.push({
                        type: &#x27;optimization&#x27;,
                        title: &#x27;Оптимизация услуг&#x27;,
                        message: &#x27;Объединение похожих услуг может снизить административные расходы&#x27;,
                        action: &#x27;Объедините услуги&#x27;
                    });
                }
                
                setAiRecommendations(recommendations);
            };

            // Добавление услуги
            const addService = () =&gt; {
                const newService = {
                    id: Date.now(),
                    type: &#x27;dosug&#x27;,
                    hours: 20,
                    level: &#x27;mittel&#x27;,
                    region: &#x27;dresden&#x27;,
                    diagnosis: &#x27;physical&#x27;
                };
                setServices([...services, newService]);
            };

            // Удаление услуги
            const removeService = (id) =&gt; {
                if (services.length &gt; 1) {
                    setServices(services.filter(s =&gt; s.id !== id));
                }
            };

            // Обновление услуги
            const updateService = (id, field, value) =&gt; {
                setServices(services.map(service =&gt; 
                    service.id === id ? {...service, [field]: value} : service
                ));
            };

            // Генерация документа
            const generateDocument = () =&gt; {
                const doc = `
ЗАЯВЛЕНИЕ НА ПЕРСОНАЛЬНЫЙ БЮДЖЕТ
===============================
Дата: ${new Date().toLocaleDateString(&#x27;de-DE&#x27;)}

УСЛУГИ:
${services.map((service, index) =&gt; {
    const calc = calculateServiceCost(service);
    return `${index + 1}. ${serviceTypes[service.type].name}
   Часов: ${service.hours} в месяц
   Регион: ${regions[service.region].name}
   Уровень: ${levels[service.level].name}
   Стоимость: ${calc.totalCost.toFixed(2)}€
   Правовое основание: ${serviceTypes[service.type].sgb}
`;
}).join(&#x27;\n&#x27;)}

ОБЩАЯ СТОИМОСТЬ: ${totalCost.toFixed(2)}€ в месяц
ГОДОВАЯ СТОИМОСТЬ: ${(totalCost * 12).toFixed(2)}€

Создано с помощью Personal Budget Calculator
                `;
                
                // Скачивание файла
                const blob = new Blob([doc], { type: &#x27;text/plain;charset=utf-8&#x27; });
                const url = URL.createObjectURL(blob);
                const a = document.createElement(&#x27;a&#x27;);
                a.href = url;
                a.download = &#x27;personal-budget-application.txt&#x27;;
                a.click();
                URL.revokeObjectURL(url);
            };

            return (
                &lt;div className=&quot;min-h-screen bg-gradient-to-br from-blue-50 to-indigo-50 py-8&quot;&gt;
                    &lt;div className=&quot;max-w-6xl mx-auto px-4&quot;&gt;
                        {/* Заголовок */}
                        &lt;div className=&quot;bg-white rounded-xl shadow-xl mb-8&quot;&gt;
                            &lt;div className=&quot;gradient-bg text-white p-6 rounded-t-xl&quot;&gt;
                                &lt;h1 className=&quot;text-3xl font-bold flex items-center&quot;&gt;
                                    &lt;span className=&quot;icon mr-3&quot;&gt;🧮&lt;/span&gt;
                                    Personal Budget Calculator
                                    &lt;span className=&quot;ml-3 px-3 py-1 bg-white/20 text-sm rounded-full&quot;&gt;
                                        Standalone Version
                                    &lt;/span&gt;
                                &lt;/h1&gt;
                                &lt;p className=&quot;mt-2 opacity-90&quot;&gt;
                                    Professional tool for German social services budgeting
                                &lt;/p&gt;
                            &lt;/div&gt;
                        &lt;/div&gt;

                        {/* AI Recommendations */}
                        {aiRecommendations.length &gt; 0 &amp;&amp; (
                            &lt;div className=&quot;bg-purple-50 border border-purple-200 rounded-lg p-6 mb-8&quot;&gt;
                                &lt;h3 className=&quot;text-lg font-semibold text-purple-800 mb-4 flex items-center&quot;&gt;
                                    &lt;span className=&quot;icon mr-2&quot;&gt;🤖&lt;/span&gt;
                                    AI Рекомендации
                                &lt;/h3&gt;
                                {aiRecommendations.map((rec, index) =&gt; (
                                    &lt;div key={index} className=&quot;bg-white p-4 rounded-lg mb-3 last:mb-0&quot;&gt;
                                        &lt;div className=&quot;flex justify-between items-start&quot;&gt;
                                            &lt;div&gt;
                                                &lt;h4 className=&quot;font-semibold text-gray-800&quot;&gt;{rec.title}&lt;/h4&gt;
                                                &lt;p className=&quot;text-gray-600 text-sm mt-1&quot;&gt;{rec.message}&lt;/p&gt;
                                                &lt;p className=&quot;text-blue-600 text-sm mt-2&quot;&gt;💡 {rec.action}&lt;/p&gt;
                                            &lt;/div&gt;
                                            &lt;span className={`px-2 py-1 text-xs rounded-full ${
                                                rec.type === &#x27;warning&#x27; ? &#x27;bg-red-100 text-red-700&#x27; : &#x27;bg-blue-100 text-blue-700&#x27;
                                            }`}&gt;
                                                {rec.type}
                                            &lt;/span&gt;
                                        &lt;/div&gt;
                                    &lt;/div&gt;
                                ))}
                            &lt;/div&gt;
                        )}

                        {/* Services List */}
                        &lt;div className=&quot;bg-white rounded-xl shadow-lg p-6 mb-8&quot;&gt;
                            &lt;div className=&quot;flex justify-between items-center mb-6&quot;&gt;
                                &lt;h2 className=&quot;text-2xl font-semibold text-gray-800&quot;&gt;Услуги и расчеты&lt;/h2&gt;
                                &lt;button 
                                    onClick={addService}
                                    className=&quot;bg-green-600 hover:bg-green-700 text-white px-4 py-2 rounded-lg font-medium flex items-center&quot;
                                &gt;
                                    &lt;span className=&quot;icon mr-2&quot;&gt;➕&lt;/span&gt;
                                    Добавить услугу
                                &lt;/button&gt;
                            &lt;/div&gt;

                            {services.map((service, index) =&gt; {

                                return (

                                                &lt;button

                                                    ✕
                                                &lt;/button&gt;
                                            )}
                                        &lt;/div&gt;

                                        &lt;div className=&quot;grid grid-cols-1 md:grid-cols-4 gap-4 mb-4&quot;&gt;
                                            &lt;div&gt;
                                                &lt;label className=&quot;block text-sm font-medium text-gray-700 mb-2&quot;&gt;
                                                    Тип услуги
                                                &lt;/label&gt;
                                                &lt;select 
                                                    value={service.type}
                                                    onChange={(e) =&gt; updateService(service.id, &#x27;type&#x27;, e.target.value)}
                                                    className=&quot;w-full p-3 border border-gray-300 rounded-lg focus:ring-2 focus:ring-blue-500 focus:border-blue-500&quot;
                                                &gt;
                                                    {Object.entries(serviceTypes).map(([key, type]) =&gt; (
                                                        &lt;option key={key} value={key}&gt;{type.name}&lt;/option&gt;
                                                    ))}
                                                &lt;/select&gt;
                                            &lt;/div&gt;

                                            &lt;div&gt;
                                                &lt;label className=&quot;block text-sm font-medium text-gray-700 mb-2&quot;&gt;
                                                    Часов в месяц
                                                &lt;/label&gt;
                                                &lt;input 
                                                    type=&quot;number&quot; 
                                                    value={service.hours}
                                                    onChange={(e) =&gt; updateService(service.id, &#x27;hours&#x27;, parseInt(e.target.value) || 0)}
                                                    className=&quot;w-full p-3 border border-gray-300 rounded-lg focus:ring-2 focus:ring-blue-500 focus:border-blue-500&quot;
                                                /&gt;
                                            &lt;/div&gt;

                                            &lt;div&gt;
                                                &lt;label className=&quot;block text-sm font-medium text-gray-700 mb-2&quot;&gt;
                                                    ICF-уровень
                                                &lt;/label&gt;
                                                &lt;select 
                                                    value={service.level}
                                                    onChange={(e) =&gt; updateService(service.id, &#x27;level&#x27;, e.target.value)}
                                                    className=&quot;w-full p-3 border border-gray-300 rounded-lg focus:ring-2 focus:ring-blue-500 focus:border-blue-500&quot;
                                                &gt;
                                                    {Object.entries(levels).map(([key, level]) =&gt; (
                                                        &lt;option key={key} value={key}&gt;{level.name}&lt;/option&gt;
                                                    ))}
                                                &lt;/select&gt;
                                            &lt;/div&gt;

                                            &lt;div&gt;
                                                &lt;label className=&quot;block text-sm font-medium text-gray-700 mb-2&quot;&gt;
                                                    Регион
                                                &lt;/label&gt;
                                                &lt;select 
                                                    value={service.region}
                                                    onChange={(e) =&gt; updateService(service.id, &#x27;region&#x27;, e.target.value)}
                                                    className=&quot;w-full p-3 border border-gray-300 rounded-lg focus:ring-2 focus:ring-blue-500 focus:border-blue-500&quot;
                                                &gt;
                                                    {Object.entries(regions).map(([key, region]) =&gt; (
                                                        &lt;option key={key} value={key}&gt;{region.name}&lt;/option&gt;
                                                    ))}
                                                &lt;/select&gt;
                                            &lt;/div&gt;
                                        &lt;/div&gt;

                                        {/* Детализация расчета */}
                                        &lt;div className=&quot;bg-gray-50 p-4 rounded-lg&quot;&gt;
                                            &lt;h4 className=&quot;font-semibold mb-3&quot;&gt;Расчет стоимости:&lt;/h4&gt;
                                            &lt;div className=&quot;grid grid-cols-2 md:grid-cols-3 gap-3 text-sm&quot;&gt;
                                                &lt;div&gt;Базовая стоимость: &lt;strong&gt;{calculation.baseCost.toFixed(2)}€&lt;/strong&gt;&lt;/div&gt;
                                                &lt;div&gt;Соцвзносы: &lt;strong&gt;{calculation.socialContributions.toFixed(2)}€&lt;/strong&gt;&lt;/div&gt;
                                                &lt;div&gt;Отпуск/больничные: &lt;strong&gt;{calculation.vacationSick.toFixed(2)}€&lt;/strong&gt;&lt;/div&gt;
                                                &lt;div&gt;Администрирование: &lt;strong&gt;{calculation.administration.toFixed(2)}€&lt;/strong&gt;&lt;/div&gt;
                                                &lt;div&gt;Постоянные расходы: &lt;strong&gt;{calculation.fixedCosts.toFixed(2)}€&lt;/strong&gt;&lt;/div&gt;
                                                &lt;div className=&quot;font-bold text-lg text-blue-600&quot;&gt;
                                                    ИТОГО: {calculation.totalCost.toFixed(2)}€
                                                &lt;/div&gt;
                                            &lt;/div&gt;
                                            &lt;div className=&quot;mt-2 text-sm text-gray-600&quot;&gt;
                                                Ставка: {calculation.hourlyRate.toFixed(2)}€/час | 
                                                Правовое основание: {serviceTypes[service.type].sgb}
                                            &lt;/div&gt;
                                        &lt;/div&gt;
                                    &lt;/div&gt;
                                );
                            })}
                        &lt;/div&gt;

                        {/* Total Summary */}
                        &lt;div className=&quot;gradient-bg text-white rounded-xl p-8 mb-8&quot;&gt;
                            &lt;h2 className=&quot;text-3xl font-bold mb-6 text-center&quot;&gt;Общий расчет бюджета&lt;/h2&gt;
                            &lt;div className=&quot;grid grid-cols-1 md:grid-cols-3 gap-6 text-center&quot;&gt;
                                &lt;div className=&quot;bg-white/20 p-6 rounded-lg&quot;&gt;
                                    &lt;div className=&quot;text-4xl font-bold mb-2&quot;&gt;{services.reduce((sum, s) =&gt; sum + s.hours, 0)}&lt;/div&gt;
                                    &lt;div className=&quot;text-lg opacity-90&quot;&gt;Часов в месяц&lt;/div&gt;
                                &lt;/div&gt;
                                &lt;div className=&quot;bg-white/20 p-6 rounded-lg&quot;&gt;
                                    &lt;div className=&quot;text-4xl font-bold mb-2&quot;&gt;{totalCost.toFixed(2)}€&lt;/div&gt;
                                    &lt;div className=&quot;text-lg opacity-90&quot;&gt;Месячная стоимость&lt;/div&gt;
                                &lt;/div&gt;
                                &lt;div className=&quot;bg-white/20 p-6 rounded-lg&quot;&gt;
                                    &lt;div className=&quot;text-4xl font-bold mb-2&quot;&gt;{(totalCost * 12).toFixed(2)}€&lt;/div&gt;
                                    &lt;div className=&quot;text-lg opacity-90&quot;&gt;Годовая стоимость&lt;/div&gt;
                                &lt;/div&gt;
                            &lt;/div&gt;
                            
                            &lt;div className=&quot;flex justify-center mt-8&quot;&gt;
                                &lt;button
                                    onClick={generateDocument}
                                    className=&quot;bg-white text-blue-600 px-8 py-3 rounded-lg font-semibold hover:bg-gray-100 flex items-center text-lg&quot;
                                &gt;
                                    &lt;span className=&quot;icon mr-2&quot;&gt;📄&lt;/span&gt;
                                    Сгенерировать заявление
                                &lt;/button&gt;
                            &lt;/div&gt;
                        &lt;/div&gt;

                        {/* Footer */}
                        &lt;div className=&quot;text-center text-gray-500 text-sm&quot;&gt;
                            &lt;p&gt;Personal Budget Calculator v3.1 - Standalone HTML Version&lt;/p&gt;
                            &lt;p className=&quot;mt-1&quot;&gt;Соответствует требованиям SGB IX и BTHG | GDPR Compliant&lt;/p&gt;
                        &lt;/div&gt;
                    &lt;/div&gt;
                &lt;/div&gt;
            );
        };

        // Рендер приложения (React 18 + фолбэк на React 17 API)
        const container = document.getElementById(&#x27;root&#x27;);
        const root = (ReactDOM.createRoot)
          ? ReactDOM.createRoot(container)
          : { render: (el) =&gt; ReactDOM.render(el, container) };

        root.render(&lt;PersonalBudgetCalculator /&gt;);
    &lt;/script&gt;
&lt;/body&gt;
&lt;/html&gt;
